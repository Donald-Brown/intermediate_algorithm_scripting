Convert HTML Entities

Convert the characters &, <, >, " (double quote), and ' (apostrophe), in a string to their corresponding HTML entities. 


  function convertHTML(str) {
    // return str
    //   .replace(/&/g, "&")
    //   .replace(//g, ">")
    //   .replace(/"/g, """).replace(/'/g, ''');
  
    let entities = {
      '&': '&',
      '<': '<',
      '>': '>',
      '"': '"',
      '\'': '''
    }
    return str.replace(/([&<>"'])/g, match => entities[match]);
  }
